# Batch Operations (transactions batch-update)

Implement batch update for transactions with concurrency control.

## Purpose
AI agents often need to update many transactions at once (e.g., recategorize all transactions from a merchant). Batch operations are more efficient than individual updates.

## Command

### `monarch transactions batch-update`
```python
@app.command("batch-update")
@handle_errors
def batch_update(
    category_id: Optional[str] = typer.Option(None, "--category", "-c"),
    notes: Optional[str] = typer.Option(None, "--notes", "-n"),
    stdin: bool = typer.Option(False, "--stdin", help="Read transaction IDs from stdin"),
    ids: Optional[list[str]] = typer.Argument(None, help="Transaction IDs"),
    max_concurrency: int = typer.Option(4, "--max-concurrency", help="Max parallel operations"),
    dry_run: bool = typer.Option(False, "--dry-run"),
    format: OutputFormat = typer.Option(OutputFormat.JSON, "--format", "-f"),
):
    """Update multiple transactions."""
    # Get transaction IDs from args or stdin
    transaction_ids = list(ids) if ids else []
    if stdin:
        for line in sys.stdin:
            line = line.strip()
            if line:
                transaction_ids.append(line)
    
    if not transaction_ids:
        raise ValidationError("No transaction IDs provided")
    
    if dry_run:
        output({
            "dry_run": True,
            "transaction_count": len(transaction_ids),
            "changes": {"category_id": category_id, "notes": notes},
        }, format)
        return
    
    # Execute with concurrency control
    results = run_async(batch_update_transactions(
        transaction_ids,
        category_id=category_id,
        notes=notes,
        max_concurrency=max_concurrency,
    ))
    output(results, format)
```

## Concurrency Control
```python
async def batch_update_transactions(
    transaction_ids: list[str],
    max_concurrency: int = 4,
    **update_kwargs,
) -> list[dict]:
    """Update multiple transactions with concurrency limit."""
    semaphore = asyncio.Semaphore(max_concurrency)
    client = get_authenticated_client()
    
    async def update_one(txn_id: str) -> dict:
        async with semaphore:
            try:
                await client.update_transaction(txn_id, **update_kwargs)
                return {"id": txn_id, "status": "success"}
            except Exception as e:
                return {"id": txn_id, "status": "error", "message": str(e)}
    
    tasks = [update_one(txn_id) for txn_id in transaction_ids]
    return await asyncio.gather(*tasks)
```

## CLI Examples
```bash
# Batch update from stdin
monarch transactions list --preset this-month --search "Starbucks" -q \
    | monarch transactions batch-update --stdin --category CAT_COFFEE

# Batch update specific IDs
monarch transactions batch-update TXN1 TXN2 TXN3 --category CAT123

# Dry run to preview
echo -e "TXN1\nTXN2" | monarch transactions batch-update --stdin --category CAT123 --dry-run

# Control concurrency
monarch transactions batch-update --stdin --category CAT123 --max-concurrency 2 < ids.txt
```

## Output Format

```json
    {"id": "TXN3", "status": "error", "message": "Not found"}
]
```
