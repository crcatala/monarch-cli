# Plain Output Format (Human-Friendly Default)

Add PLAIN output format with TTY-aware defaults. This ensures consistency with auth commands (which use styled plain text) and follows CLI best practices.

## Background

Current state:
- Auth commands: Use `console.print()` with styled plain text (good!)
- Data commands: `output()` defaults to JSON (inconsistent!)

Desired state:
- All commands: Plain/styled text by default in terminal
- When piped: JSON for machine parsing
- Explicit `--json` or `--format json`: Always JSON

## Location
- `src/monarch_cli/output/__init__.py` - Add PLAIN format and TTY-aware default
- `src/monarch_cli/output/plain.py` - New file for plain formatter

## Implementation

### 1. Add PLAIN to OutputFormat enum

```python
class OutputFormat(str, Enum):
    """Output format options."""
    JSON = "json"
    PLAIN = "plain"  # NEW: Human-friendly styled output
    TABLE = "table"
    CSV = "csv"
    COMPACT = "compact"
```

### 2. TTY-aware default format

```python
def get_default_format() -> OutputFormat:
    """Get default format based on TTY detection.
    
    - Terminal (TTY): PLAIN for human-readable output
    - Piped/redirected: JSON for machine parsing
    """
    if sys.stdout.isatty():
        return OutputFormat.PLAIN
    return OutputFormat.JSON
```

### 3. Color-aware utilities

```python
def should_use_color() -> bool:
    """Determine if colored output should be used.
    
    Colors disabled when:
    - --no-color flag passed (check global state)
    - NO_COLOR env var set (any non-empty value)
    - TERM=dumb
    - stdout is not a TTY
    """
    if _no_color_flag:
        return False
    if os.environ.get("NO_COLOR"):
        return False
    if os.environ.get("TERM") == "dumb":
        return False
    if not sys.stdout.isatty():
        return False
    return True
```

### 4. Plain formatter (src/monarch_cli/output/plain.py)

Based on raindrop-cli pattern:

```python
"""Plain text output formatter with styled labels."""

from typing import Any
from rich.console import Console
from rich.text import Text

# Field icons for visual scanning
FIELD_ICONS = {
    "id": "🔖",
    "name": "📌",
    "balance": "💰",
    "type": "📁",
    "institution": "🏦",
    "date": "📅",
    "amount": "💵",
    "category": "🏷️",
    "description": "📝",
}

def get_field_icon(key: str) -> str:
    """Get emoji icon for a field."""
    return FIELD_ICONS.get(key.lower(), "•")

def format_plain_value(value: Any) -> str | None:
    """Format a value for plain output. Returns None if empty."""
    if value is None or value == "":
        return None
    if isinstance(value, list):
        return ", ".join(str(v) for v in value) if value else None
    return str(value)

def format_plain(data: Any, columns: list[str] | None = None) -> str:
    """Format data as styled plain text.
    
    Args:
        data: Dict or list of dicts to format
        columns: Optional list of keys to include (in order)
    
    Returns:
        Formatted string with icons, labels, and values
    """
    items = data if isinstance(data, list) else [data]
    
    if not items:
        return "No results found."
    
    # Use provided columns or infer from first item
    if columns is None and items:
        columns = list(items[0].keys())
    
    lines = []
    for i, item in enumerate(items):
        if i > 0:
            lines.append("")
            lines.append("─" * 50)
            lines.append("")
        
        for key in columns:
            if key not in item:
                continue
            icon = get_field_icon(key)
            value = format_plain_value(item.get(key))
            label = key.replace("_", " ").title()
            
            if value is None:
                lines.append(f"{icon} {label}: —")
            else:
                lines.append(f"{icon} {label}: {value}")
    
    return "\n".join(lines)


def print_plain(data: Any, columns: list[str] | None = None) -> None:
    """Print data in plain format to stdout."""
    console = Console(highlight=False)
    text = format_plain(data, columns)
    
    if should_use_color():
        # Add styling for terminal
        console.print(text)
    else:
        # Plain text without ANSI codes
        print(text)
```

### 5. Update output() function

```python
def output(
    data: Any,
    format: OutputFormat | None = None,  # None = use default
    ndjson: bool = False,
    raw: bool = False,
) -> None:
    """Output data in specified format.
    
    Args:
        data: Data to output
        format: Output format. None = TTY-aware default (plain/json)
        ndjson: Stream as newline-delimited JSON
        raw: Pass through without formatting
    """
    if raw:
        print(data)
        return
    
    if ndjson:
        for item in (data if isinstance(data, list) else [data]):
            print(json.dumps(item, default=str))
        return
    
    # Use TTY-aware default if format not specified
    effective_format = format if format is not None else get_default_format()
    
    if effective_format == OutputFormat.PLAIN:
        print_plain(data)
    elif effective_format == OutputFormat.JSON:
        print(json.dumps(data, indent=2, default=str))
    elif effective_format == OutputFormat.TABLE:
        print_table(data)
    elif effective_format == OutputFormat.CSV:
        print_csv(data)
    elif effective_format == OutputFormat.COMPACT:
        print(json.dumps(data, default=str))
```
