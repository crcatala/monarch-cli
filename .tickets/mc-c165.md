## test_accounts.py
```python
import pytest
from monarch_cli.transformers.accounts import transform_account, transform_accounts


class TestTransformAccount:

    def test_extracts_required_fields(self):
        raw = {
            "id": "ACC123",
            "displayName": "My Checking",
            "currentBalance": 5000.00,
            "type": {"name": "checking", "display": "Checking"},
            "subtype": {"display": "Personal"},
            "institution": {"name": "Chase"},
            "isHidden": False,
            "isManual": False,
            "updatedAt": "2024-01-15T10:00:00Z",
        }

        result = transform_account(raw)

        assert result["id"] == "ACC123"
        assert result["name"] == "My Checking"
        assert result["balance"] == 5000.00
        assert result["type"] == "Checking"
        assert result["institution"] == "Chase"
        assert result["is_active"] is True
        assert result["is_manual"] is False

    def test_inverts_is_hidden_to_is_active(self):
        raw = {"isHidden": True, "id": "1", "displayName": "Hidden"}
        result = transform_account(raw)
        assert result["is_active"] is False

    def test_handles_missing_optional_fields(self):
        raw = {"id": "1", "displayName": "Minimal"}
        result = transform_account(raw)
        assert result["type"] is None
        assert result["institution"] is None


class TestTransformAccounts:

    def test_transforms_list(self):
        raw = {
            "accounts": [
                {"id": "1", "displayName": "A"},
                {"id": "2", "displayName": "B"},
            ]
        }

        result = transform_accounts(raw)

        assert len(result) == 2
        assert result[0]["id"] == "1"
        assert result[1]["id"] == "2"

    def test_handles_empty_list(self):
        raw = {"accounts": []}
        result = transform_accounts(raw)
        assert result == []

    def test_handles_missing_accounts_key(self):
        raw = {}
        result = transform_accounts(raw)
        assert result == []
```

## test_transactions.py
```python
import pytest
from monarch_cli.transformers.transactions import transform_transaction, transform_transactions


class TestTransformTransaction:

    def test_extracts_required_fields(self):
        raw = {
            "id": "TXN123",
            "date": "2024-01-15",
            "amount": -42.50,
            "merchant": {"name": "Coffee Shop"},
            "plaidName": "COFFEE SHOP #123",
            "category": {"id": "CAT1", "name": "Food & Drink"},
            "account": {"id": "ACC1", "displayName": "Checking"},
            "pending": False,
            "notes": "Morning coffee",
        }

        result = transform_transaction(raw)

        assert result["id"] == "TXN123"
        assert result["date"] == "2024-01-15"
        assert result["amount"] == -42.50
        assert result["description"] == "Coffee Shop"  # Uses merchant.name
        assert result["category"] == "Food & Drink"
        assert result["category_id"] == "CAT1"
        assert result["is_pending"] is False

    def test_falls_back_to_plaid_name(self):
        raw = {
            "id": "1",
            "merchant": {},  # No name
            "plaidName": "RAWMERCHANT",
        }
        result = transform_transaction(raw)
        assert result["description"] == "RAWMERCHANT"

    def test_handles_pending_transaction(self):
        raw = {"id": "1", "pending": True}
        result = transform_transaction(raw)
        assert result["is_pending"] is True
```
